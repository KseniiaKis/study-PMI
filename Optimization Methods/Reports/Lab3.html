ax.set_zlabel('Z axis')

# Символьный расчет градиента с помощью sympy
x_sym, y_sym = sp.symbols('x y')
f_sym = sp.sin(x_sym) + sp.cos(y_sym)
gradient = [sp.diff(f_sym, var) for var in (x_sym, y_sym)]
print("Градиент функции:")
for g in gradient:
    print(g)

cos(x)
-sin(y)
